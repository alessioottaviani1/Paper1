%% =============================================================================
%% EDHEC FORMAT: PAGINATION
%% =============================================================================
%% "The numeral pagination is placed one half inch up from the bottom of the page"
%% "Page numbers are placed in the lower right hand corner"
%%
%% With bottom margin = 1in, footskip controls distance from text to footer.
%% Default footskip ~ 30pt. We set it so page number sits at 0.5in from bottom.
%% 1in margin - 0.5in = 0.5in from bottom of text block to page number.

\setlength{\footskip}{0.5in}

\fancypagestyle{plain}{
  \fancyhf{}
  \renewcommand{\headrulewidth}{0pt}
  \fancyfoot[R]{\thepage}
}
\pagestyle{plain}


%% =============================================================================
%% EDHEC FORMAT: SPACING
%% =============================================================================
%% "Spacing throughout body of text: double spacing"
%% "Include one space after sentence punctuation" (not two)
%% "No pages end with headings and subheadings"
%% "No paragraphs end with one line at the top of a page"

\doublespacing
\frenchspacing                            % One uniform space after periods (EDHEC: "one space")

%% Prevent widows and orphans (EDHEC: "No paragraphs end with one line at top of page")
\widowpenalty=10000
\clubpenalty=10000


%% =============================================================================
%% EDHEC FORMAT: TOC, LOT, LOF SPACING
%% =============================================================================
%% "Spacing for table of contents, list of tables, list of figures: single spacing may be used"

\setlength{\cftbeforesecskip}{3pt}        % Tighter TOC spacing
\setlength{\cftbeforesubsecskip}{1pt}


%% =============================================================================
%% EDHEC FORMAT: BIBLIOGRAPHY SPACING
%% =============================================================================
%% "Spacing for footnotes and bibliographies: single spacing within each entry
%%  but double spacing between each entry"

\setlength{\bibsep}{1.0\baselineskip}     % Double spacing between bib entries
